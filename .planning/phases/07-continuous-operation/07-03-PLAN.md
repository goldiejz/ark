```bash
# continuous_health_monitor
#   Scans portfolio for stuck phases. Returns 0; all signals are audit rows.
#   For each project under $ARK_PORTFOLIO_ROOT:
#     - Read .planning/STATE.md current_phase value
#     - mtime check: STATE.md modified > 24h ago

#       append ESCALATIONS.md entry + log STUCK_ESCALATED.

# continuous_auto_pause_check
#   Inspects last 3 TICK_COMPLETE audit rows. If all 3 have non-zero
#   FAILED count → create PAUSE file, log AUTO_PAUSED, escalate.
#   Idempotent (no-op if PAUSE already exists).
```

       - PORTFOLIO_ROOT="${ARK_PORTFOLIO_ROOT:-$HOME/code}"
       - For each `proj_dir` matching `$PORTFOLIO_ROOT/*/.planning/STATE.md`:
         - slug=basename of parent dir
         - current_phase=`grep -m1 '^current_phase:' "$proj_dir" | awk -F'"' '{print $2}'` (fallback awk for unquoted)

         - now=`date +%s`; age=$((now - state_mtime))
         - stuck_mtime=$([[ $age -gt 86400 ]] && echo 1 || echo 0)
         - last_commit_epoch=$(cd "$(dirname "$(dirname "$proj_dir")")" && git log -1 --format=%ct 2>/dev/null) — empty means no commits or not a repo; treat as stuck_git=1 if empty OR (now - last_commit_epoch) > 86400
         - if [[ $stuck_mtime -eq 1 && $stuck_git -eq 1 ]]:
           - corr_id="$slug:$current_phase"
           - `_policy_log "continuous" "STUCK_PHASE_DETECTED" "age:$age" "{\"slug\":\"$slug\",\"phase\":\"$current_phase\"}" "$corr_id"`
           - count=$(sqlite3 "$VAULT_PATH/observability/policy.db" "SELECT COUNT(*) FROM decisions WHERE class='continuous' AND decision='STUCK_PHASE_DETECTED' AND correlation_id='$corr_id' AND ts >= datetime('now','-60 minutes')")

           - if [[ $count -ge 3 && $escalated_recently -eq 0 ]]:
             - append ESCALATIONS.md entry (atomic tmp+mv) — markdown block with slug, phase, age, decision_id chain
             - `_policy_log "continuous" "STUCK_ESCALATED" "consecutive:$count" "{\"slug\":\"$slug\",\"phase\":\"$current_phase\"}" "$corr_id"`

         - `_policy_log "continuous" "AUTO_PAUSED" "consecutive_failures:3" ...`
