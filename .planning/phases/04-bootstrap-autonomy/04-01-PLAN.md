    ```bash
    # bootstrap_customer_dir <customer> â€” echoes the customer dir path; does NOT create.
    # 04-05 adds mkdir-lock creation. 04-01 stub: echo "$VAULT_PATH/customers/$customer"
    bootstrap_customer_dir() {
      local customer="$1"
      [[ -z "$customer" ]] && return 1
      local vault="${ARK_HOME:-$HOME/vaults/ark}"
      echo "$vault/customers/$customer"
    }

    # bootstrap_customer_resolve_policy <customer> <dotted.key> [default]
    # Reads <customer-dir>/policy.yml for <dotted.key>; falls back to default.
    # 04-05 wires this into the cascading config in policy-config.sh.
    # 04-01 stub: if file exists, awk-grep the key; else echo default.
    bootstrap_customer_resolve_policy() {
      local customer="$1" key="$2" default_value="${3:-}"
      [[ -z "$customer" ]] && { echo "$default_value"; return 0; }
      local file
      file="$(bootstrap_customer_dir "$customer")/policy.yml"
      [[ ! -f "$file" ]] && { echo "$default_value"; return 0; }
      # Reuse the same awk pattern as policy-config.sh::_pc_read_yaml_key.
      # Inline copy â€” small enough to not warrant cross-sourcing.
      local val
      val=$(awk -v k="$key" '
        /^[[:space:]]*#/ { next }
        /^[[:space:]]*$/ { next }
        {
          sub(/^[[:space:]]+/, "")
          idx = index($0, ":")
          if (idx == 0) next
          key = substr($0, 1, idx-1)
          val = substr($0, idx+1)
          sub(/^[[:space:]]+/, "", val)
          sub(/[[:space:]]+$/, "", val)
          if (key == k) { print val; exit 0 }
        }
      ' "$file")
      [[ -z "$val" ]] && val="$default_value"
      echo "$val"
    }

    # bootstrap_customer_init <customer>
    # 04-05 implements mkdir-lock + idempotent customer dir + seed policy.yml.
    # 04-01 stub: just `mkdir -p` the dir if missing. Idempotent.
    bootstrap_customer_init() {
      local customer="$1"
      [[ -z "$customer" ]] && return 1
      local dir
      dir="$(bootstrap_customer_dir "$customer")"
      mkdir -p "$dir"
      return 0
    }
    ```
